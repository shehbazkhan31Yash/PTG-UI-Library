# PtgUiModal Component

The `PtgUiModal` component is a versatile and customizable modal dialog for displaying content, actions, or confirmation dialogs. It supports options for a header, footer, and configurable buttons.

---

## **Usage**
```tsx
import React, { useState } from 'react';
import { PtgUiModal } from './modal';
import { PtgUiButton } from '../button/button';

const ExampleModal = () => {
  const [isOpen, setIsOpen] = useState(false);

  const handleModalClose = () => {
    setIsOpen(false);
  };

  const handleModalConfirm = () => {
    alert('Confirmed!');
    setIsOpen(false);
  };

  return (
    <>
      <PtgUiButton
        text="Open Modal"
        onClick={() => setIsOpen(true)}
        textColor="#fff"
        backgroundColor="#007bff"
      />
      <PtgUiModal
        isOpen={isOpen}
        modalSize="md"
        showHeader={true}
        header="Example Modal Header"
        showFooter={true}
        confirmButton="Confirm"
        cancelButton="Cancel"
        content="This is an example modal."
        onModalClose={handleModalClose}
        onConfirmed={handleModalConfirm}
      />
    </>
  );
};
